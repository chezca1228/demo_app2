def to_s(parent = false)
  if bracketed?
    if !parent && ignorecase
      "(?i-mx:[#{negate && '^'}#{value}])#{quantifier}"
    else
      "[#{negate && '^'}#{value}]#{quantifier}"
    end
  else
    super
  end
end
