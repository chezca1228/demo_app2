def shell_commands( cmd, *args )
  cmdline = [ cmd.shellsplit ]
  cmdline << args.flatten.collect { |a| a.to_s.gsub(/([&|()<>^])/, "^\\1") }
  return commandline_normalize( cmdline )
end
